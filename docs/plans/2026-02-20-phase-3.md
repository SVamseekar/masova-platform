If there's no analytics API slice, create `frontend/src/store/api/analyticsApi.ts`:

```typescript
import { createApi, fetchBaseQuery } from '@reduxjs/toolkit/query/react';

export interface SalesMetrics {
  totalRevenue: number;
  totalOrders: number;
  averageOrderValue: number;
  period: string;
}

export interface SalesTrend {
  date: string;
  revenue: number;
  orders: number;
}

export interface TopProduct {
  name: string;
  quantity: number;
  revenue: number;
}

export const analyticsApi = createApi({
  reducerPath: 'analyticsApi',
  baseQuery: fetchBaseQuery({
    baseUrl: '/api/analytics',
    prepareHeaders: (headers) => {
      const token = localStorage.getItem('token');
      if (token) headers.set('Authorization', `Bearer ${token}`);
      return headers;
    },
  }),
  endpoints: (builder) => ({
    getSalesMetrics: builder.query<SalesMetrics, { storeId?: string; period?: string }>({
      query: (params) => ({ url: '/sales-metrics', params }),
    }),
    getSalesTrend: builder.query<SalesTrend[], { storeId?: string; days?: number }>({
      query: (params) => ({ url: '/sales-trend', params }),
    }),
    getTopProducts: builder.query<TopProduct[], { storeId?: string; limit?: number }>({
      query: (params) => ({ url: '/top-products', params }),
    }),
  }),
});

export const { useGetSalesMetricsQuery, useGetSalesTrendQuery, useGetTopProductsQuery } = analyticsApi;
```

Register in store if `analyticsApi` is new — check `frontend/src/store/store.ts` and add reducer + middleware.

**Step 3: Create AnalyticsDashboard.tsx**

```typescript
import React from 'react';
import { Box, Grid, Paper, Typography, CircularProgress } from '@mui/material';
import {
  LineChart, Line, XAxis, YAxis, CartesianGrid, Tooltip, ResponsiveContainer,
  BarChart, Bar, Legend,
} from 'recharts';
import { useGetSalesMetricsQuery, useGetSalesTrendQuery, useGetTopProductsQuery } from '../../store/api/analyticsApi';

const StatCard: React.FC<{ title: string; value: string | number; subtitle?: string }> = ({ title, value, subtitle }) => (
  <Paper sx={{ p: 3, textAlign: 'center' }}>
    <Typography variant="h4" fontWeight="bold" color="primary">{value}</Typography>
    <Typography variant="subtitle1" color="text.secondary">{title}</Typography>
    {subtitle && <Typography variant="caption" color="text.disabled">{subtitle}</Typography>}
  </Paper>
);

const AnalyticsDashboard: React.FC = () => {
  const { data: metrics, isLoading: metricsLoading } = useGetSalesMetricsQuery({ period: 'week' });
  const { data: trend = [] } = useGetSalesTrendQuery({ days: 7 });
  const { data: topProducts = [] } = useGetTopProductsQuery({ limit: 5 });

  if (metricsLoading) return <Box display="flex" justifyContent="center" mt={8}><CircularProgress /></Box>;

  return (
    <Box p={3}>
      <Typography variant="h5" fontWeight="bold" gutterBottom>Analytics Dashboard</Typography>

      {/* KPI Cards */}
      <Grid container spacing={3} mb={4}>
        <Grid item xs={12} sm={4}>
          <StatCard title="Total Revenue" value={`₹${metrics?.totalRevenue?.toFixed(0) ?? 0}`} subtitle="This week" />
        </Grid>
        <Grid item xs={12} sm={4}>
          <StatCard title="Total Orders" value={metrics?.totalOrders ?? 0} subtitle="This week" />
        </Grid>
        <Grid item xs={12} sm={4}>
          <StatCard title="Avg Order Value" value={`₹${metrics?.averageOrderValue?.toFixed(0) ?? 0}`} />
        </Grid>
      </Grid>

      {/* Revenue trend line chart */}
      <Paper sx={{ p: 3, mb: 3 }}>
        <Typography variant="h6" gutterBottom>Revenue Trend (7 days)</Typography>
        <ResponsiveContainer width="100%" height={250}>
          <LineChart data={trend}>
            <CartesianGrid strokeDasharray="3 3" />
            <XAxis dataKey="date" />
            <YAxis />
            <Tooltip />
            <Line type="monotone" dataKey="revenue" stroke="#D32F2F" strokeWidth={2} dot={false} />
          </LineChart>
        </ResponsiveContainer>
      </Paper>

      {/* Top products bar chart */}
      <Paper sx={{ p: 3 }}>
        <Typography variant="h6" gutterBottom>Top Products</Typography>
        <ResponsiveContainer width="100%" height={250}>
          <BarChart data={topProducts}>
            <CartesianGrid strokeDasharray="3 3" />
            <XAxis dataKey="name" />
            <YAxis />
            <Tooltip />
            <Legend />
            <Bar dataKey="quantity" fill="#D32F2F" name="Units Sold" />
          </BarChart>
        </ResponsiveContainer>
      </Paper>
    </Box>
  );
};

export default AnalyticsDashboard;
```
